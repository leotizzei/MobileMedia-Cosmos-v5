41      private String[] convertListToArray(Enumeration stringEnum){
42          Vector stringVector = new Vector();
43          for (Enumeration iter = stringEnum; iter.hasMoreElements();) {
44              String element = (String) iter.nextElement();
45              stringVector.addElement(element);
46          }
47          
48          String[] stringArray = new String[stringVector.size()];
49          for (int i=0; i < stringVector.size(); i++){
50              stringArray[i] = (String) stringVector.elementAt(i);
51          }
52          return stringArray;
53      }
