3   package br.unicamp.ic.sed.mobilemedia.sms.impl;
4   
5   import javax.microedition.lcdui.Command;
6   import javax.microedition.midlet.MIDlet;
7   
8   import main.ControllerInterface;
9   import br.unicamp.ic.sed.mobilemedia.sms.spec.prov.IManager;
10  import br.unicamp.ic.sed.mobilemedia.sms.spec.prov.ISms;
11  import br.unicamp.ic.sed.mobilemedia.sms.spec.req.IMobileResources;
12  
13  public class ISmsFacade implements ISms{
14  
15      private SmsSenderController senderController = null;
16      private ControllerInterface nextController;
17      
18      
19      public ISmsFacade(){
20          IManager manager = ComponentFactory.createInstance();
21          IMobileResources mobile = (IMobileResources) manager.getRequiredInterface("IMobileResources");
22          MIDlet midlet = mobile.getMainMIDlet();
23          
24          senderController = new SmsSenderController( midlet );
25          senderController.setNextController(nextController);
26          
27          SmsReceiverController controller = new SmsReceiverController( midlet );
28          //controller.setNextController(albumController);
29          SmsReceiverThread smsR = new SmsReceiverThread(midlet , controller);
30          System.out.println("SmsController::Starting SMSReceiver Thread");
31          new Thread(smsR).start();
32      }
33      
34      public boolean postCommand(Command command) {
35          if( senderController.postCommand(command))
36              return true;
37          else if( nextController != null )
38              return nextController.postCommand(command);
39          return false;
40      }
41  
42      public ControllerInterface getNextController() {
43          return nextController;
44      }
45      
46      public void setNextController(ControllerInterface controller) {
47          nextController = controller;        
48      }
49  }
