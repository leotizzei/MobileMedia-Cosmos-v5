17  /**
18   * @author tyoung
19   *
20   * This interface must be implemented by all sub-Controllers used by MobilePhoto.
21   * It is for the Chain of Responsibility design pattern, and defined two methods.
22   * The entry point to handle a command using the implementing controller is the 
23   * postCommand() method. postCommand() calls handleCommand, which will return True 
24   * if the current controller has handled the command. handleCommand will return false otherwise.
25   * If handleCommand returns false, then postCommand will attempt to call the next controller
26   * in the chain, if one exists.
27   */
28  interface ControllerInterface {
29  
30      public boolean postCommand(Command command) throws PersistenceMechanismException, InvalidPhotoAlbumNameException  ;
31      
32      public boolean handleCommand(Command command) throws PersistenceMechanismException, InvalidPhotoAlbumNameException; 
33      
34      
35      
36  }
