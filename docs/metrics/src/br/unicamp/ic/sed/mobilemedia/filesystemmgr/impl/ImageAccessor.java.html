203      * Define a new photo album for mobile photo users. This creates a new
204      * record store to store photos for the album.
205      * @throws PersistenceMechanismException 
206      * @throws InvalidPhotoAlbumNameException 
207      */
208     protected void createNewPhotoAlbum(String albumName) throws PersistenceMechanismException, InvalidPhotoAlbumNameException {
209 
210         RecordStore newAlbumRS = null;
211         RecordStore newAlbumInfoRS = null;
212         if( albumName == null){
213             System.err.println("Error in class "+this.getClass().getName()+"! AlbumName is null");
214             return;
215         }
216         if (albumName.equals("")){
217             throw new InvalidPhotoAlbumNameException( "Invalid album name!" );
218 
219         }
220         
221         //check if there is an album with the same name
222         String[] albumNames  = this.getAlbumNames();
223         for (int i = 0; i < albumNames.length; i++) {
224             if (albumNames[i].equals(albumName))
225                 throw new InvalidPhotoAlbumNameException( "Invalid album name!" );
226         }
227 
228         try {
229             newAlbumRS = RecordStore.openRecordStore(ALBUM_LABEL + albumName,true);
230             newAlbumInfoRS = RecordStore.openRecordStore(INFO_LABEL + albumName, true);
231             newAlbumRS.closeRecordStore();
232             newAlbumInfoRS.closeRecordStore();
233         } catch (RecordStoreException rse) {
234             throw new PersistenceMechanismException(rse);
235         }
236

336     /**
337      * Load all existing photo albums that are defined in the record store.
338      * 
339      * @throws InvalidImageDataException
340      * @throws PersistenceMechanismException
341      */
342     protected void loadAlbums() throws InvalidImageDataException,
343     PersistenceMechanismException {
344             
345         
346         
347         // Try to find any existing Albums (record stores)
348 
349         String[] currentStores = RecordStore.listRecordStores();
350 
351         if (currentStores != null) {
352             System.out.println("ImageAccessor::loadAlbums: Found: "
353                     + currentStores.length + " existing record stores");
354             //AlbumData albumData = this.getAlbumData();
355         
356             albumData.existingRecords = true;
357             String[] temp = new String[currentStores.length];
358             int count = 0;
359 
360             // Only use record stores that follow the naming convention defined
361             for (int i = 0; i < currentStores.length; i++) {
362                 String curr = currentStores[i];
363         
364                 // If this record store is a photo album...
365                 if (curr.startsWith(ALBUM_LABEL)) {
366 
367                     // Strip out the mpa- identifier
368                     curr = curr.substring(4);
369                     // Add the album name to the array
370                     temp[i] = curr;
371                     count++;
372         
373                 }
374             }
375 
376             // Re-copy the contents into a smaller array now that we know the
377             // size
378             String[] albumNames = new String[count];
379             int count2 = 0;
380             for (int i = 0; i < temp.length; i++) {
381                 if (temp[i] != null) {
382                     albumNames[count2] = temp[i];
383                     count2++;
384                 }
385                 
386             }
387             this.setAlbumNames( albumNames );
388             
389         } else {
390             resetImageRecordStore();
391             loadAlbums();
392         }
393     
394     }

420     /**
421      * This will populate the imageInfo hashtable with the ImageInfo object,
422      * referenced by label name and populate the imageTable hashtable with Image
423      * objects referenced by the RMS record Id
424      * 
425      * @throws PersistenceMechanismException
426      */
427     protected ImageData[] loadImageDataFromRMS(String recordName)
428     throws PersistenceMechanismException, InvalidImageDataException {
429 
430         Vector imagesVector = new Vector();
431 
432         try {
433 
434             // [EF] not used            String storeName = ImageAccessor.ALBUM_LABEL + recordName;
435             String infoStoreName = ImageAccessor.INFO_LABEL + recordName;
436 
437             RecordStore infoStore = RecordStore.openRecordStore(infoStoreName,
438                     false);
439             RecordEnumeration isEnum = infoStore.enumerateRecords(null, null,
440                     false);
441 
442             while (isEnum.hasNextElement()) {
443                 // Get next record
444                 int currentId = isEnum.nextRecordId();
445                 byte[] data = infoStore.getRecord(currentId);
446 
447                 // Convert the data from a byte array into our ImageData
448                 // (metadata) object
449                 ImageUtil converter = new ImageUtil();
450                 ImageData iiObject = converter.getImageInfoFromBytes(data);
451 
452                 //              System.out.println("<* ImageAccessor.loadImageDataFromRMS *> iiObject = "+iiObject.getImageLabel());
453 
454                 // Add the info to the metadata hashtable
455                 String label = iiObject.getImageLabel();
456                 imagesVector.addElement(iiObject);
457                 AlbumData albumData = this.getAlbumData();
458                 albumData.getImageInfoTable().put(label, iiObject);
459 
460             }
461 
462             infoStore.closeRecordStore();
463 
464         }catch (RecordStoreException rse) {
465             throw new PersistenceMechanismException(rse);
466         }
467 
468         // Re-copy the contents into a smaller array
469         ImageData[] labelArray = new ImageData[imagesVector.size()];
470         imagesVector.copyInto(labelArray);
471         
472         return labelArray;
473     }

575     /**
576      * Update the Image metadata associated with this named photo
577      * @throws InvalidImageDataException 
578      * @throws PersistenceMechanismException 
579      */
580     protected boolean updateImageInfo(IImageData oldData, IImageData newData) throws InvalidImageDataException, PersistenceMechanismException {
581 
582         boolean success = false;
583         RecordStore infoStore = null;
584         try {
585 
586             // Parse the Data store name to get the Info store name
587             String infoStoreName = oldData.getParentAlbumName();
588             infoStoreName = ImageAccessor.INFO_LABEL
589             + infoStoreName.substring(ImageAccessor.ALBUM_LABEL
590                     .length());
591             infoStore = RecordStore.openRecordStore(infoStoreName, false);
592 
593             ImageUtil converter = new ImageUtil();
594             byte[] imageDataBytes = converter.getBytesFromImageInfo(newData);
595 
596             infoStore.setRecord(oldData.getRecordId(), imageDataBytes, 0,
597                     imageDataBytes.length);
598 
599         } catch (RecordStoreException rse) {
600             throw new PersistenceMechanismException(rse);
601         }
602 
603         // Update the Hashtable 'cache'
604         this.setImageInfo(oldData.getImageLabel(), newData);
605 
606         try {
607             infoStore.closeRecordStore();
608         } catch (RecordStoreNotOpenException e) {
609             //No problem if the RecordStore is not Open
610         } catch (RecordStoreException e) {
611             throw new PersistenceMechanismException( e.getMessage() );
612         }
613 
614         return success;
615     }
