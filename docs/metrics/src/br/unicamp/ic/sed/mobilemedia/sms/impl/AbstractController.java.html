58       * @see ubc.midp.mobilephoto.core.ui.controller.ControllerInterface#postCommand(javax.microedition.lcdui.Command, javax.microedition.lcdui.Displayable)

60      public boolean postCommand(Command command)  {
61          System.out.println("AbstractController::postCommand - Current controller is: " + this.getClass().getName());
62          System.out.println("Command label="+command.getLabel());
63          //If the current controller cannot handle the command, pass it to the next
64          //controller in the chain.
65  
66          try {
67              if (handleCommand(command) == false) {
68                  ControllerInterface next = this.getNextController();
69                  if (next != null) {
70                      System.out.println("Passing to next controller in chain: " + next.getClass().getName());
71                      return next.postCommand(command);
72                  } else {
73                      System.out.println("AbstractController::postCommand - Reached top of chain. No more handlers for command: " + command.getLabel());
74                      IManager manager = ComponentFactory.createInstance();
75                      
76                      //TODO call the next component to handle the command
77                      return false;
78  
79                  }
80              }
81          } catch (ImageNotFoundException e) {
82              // TODO call exception handler
83              e.printStackTrace();
84          } catch (NullAlbumDataReference e) {
85              // TODO Auto-generated catch block
86              e.printStackTrace();
87          } catch (PersistenceMechanismException e) {
88              // TODO Auto-generated catch block
89              e.printStackTrace();
90          }
91  
92          return false;
93  
94      }
