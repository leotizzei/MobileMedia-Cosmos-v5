50      public boolean postCommand(Command command) {
51          System.out.println("AbstractController::postCommand - Current controller is: " + this.getClass().getName());
52          System.out.println("Command label="+command.getLabel());
53          //If the current controller cannot handle the command, pass it to the next
54          //controller in the chain.
55          if (handleCommand(command) == false) {
56              ControllerInterface next = this.getNextController();
57              if (next != null) {
58                  System.out.println("Passing to next controller in chain: " + next.getClass().getName());
59                  return next.postCommand(command);
60              } else {
61                  System.out.println("AbstractController::postCommand - Reached top of chain. No more handlers for command: " + command.getLabel());
62                  return false;
63              }
64          }
65          return false;
66      }

75       */
76      public void commandAction(Command c, Displayable d) {
