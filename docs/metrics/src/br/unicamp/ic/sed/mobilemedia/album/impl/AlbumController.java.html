46      public boolean handleCommand(Command command) throws PersistenceMechanismException, InvalidPhotoAlbumNameException {
47          String label = command.getLabel();
48          System.out.println( "<*"+this.getClass().getName()+".handleCommand() *> " + label);
49  
50          IManager manager = ComponentFactory.createInstance();
51          IFilesystem filesystem = (IFilesystem) manager.getRequiredInterface("IFilesystem");
52          System.out.println("[AlbumController] filesystem = "+filesystem);
53          if (label.equals("Reset")) {
54              System.out.println("<* BaseController.handleCommand() *> Reset Photo Album");           
55              resetImageData();
56              ScreenSingleton.getInstance().setCurrentScreenName(Constants.ALBUMLIST_SCREEN);
57              this.goToPreviousScreen();
58              return true;
59              /** Case: Create PhotoAlbum **/
60          }else if (label.equals("New Photo Album")) {
61              ScreenSingleton.getInstance().setCurrentScreenName(Constants.NEWALBUM_SCREEN);
62              NewLabelScreen newLabelScreen =  new NewLabelScreen("Add new Photo Album", NewLabelScreen.NEW_ALBUM);
63              newLabelScreen.setCommandListener(this);
64              this.setCurrentScreen(newLabelScreen);
65              newLabelScreen = null;
66              return true;
67              /** Case: Delete Album Photo**/
68          }else if (label.equals("Delete Album")) {
69              this.deletePhotoAlbum();
70              return true;    
71              /**
72               *  TODO [EF] I think this confirmation questions are complicating the implementation
73               *  [EF] How do you know that "Yes - Delete" is to delete Photo Album instead of Photo?
74               *  Case: Yes delete Photo Album  **/
75          }else if (label.equals("Yes - Delete")) {
76  
77              //delete photo album
78              //begin - modified in MobileMedia-Cosmos-OO-v4
79  
80              String albumName = ScreenSingleton.getInstance().getCurrentStoreName();
81              filesystem.deletePhotoAlbum(albumName);
82              //end - modified in MobileMedia-Cosmos-OO-v4
83  
84              goToPreviousScreen();
85              return true;    
86              /** 
87               * [EF] Same question. How do you know that "No - Delete" is to delete Photo Album instead of Photo?
88               * Case: No delete Photo Album **/
89          }else if (label.equals("No - Delete")) {
90              goToPreviousScreen();
91              return true;    
92              /** 
93               * [EF] Again, see [EF] comments above.
94               * Case: Save new Photo Album  **/
95          } else if (label.equals("Save new label")) {
96  
97              Displayable displayable = getCurrentScreen();
98              System.out.println("[AlbumController] displayable = "+displayable);
99              if ( displayable instanceof NewLabelScreen) {
100                 //begin - modified in MobileMedia-Cosmos-OO-v4
101                 NewLabelScreen currentScreen = (NewLabelScreen)getCurrentScreen();
102                 if (currentScreen.getFormType() == NewLabelScreen.NEW_ALBUM){
103                     String albumName = currentScreen.getLabelName();
104                     System.out.println("[AlbumController] filesystem = "+filesystem);
105                     filesystem.createNewPhotoAlbum(albumName);
106                 }
107                 else{
108                     if (currentScreen.getFormType() == NewLabelScreen.LABEL_PHOTO) {
109                         //do nothing?
110                     }
111                 }
112                 //end - modified in MobileMedia-Cosmos-OO-v4
113             }
114 
115             goToPreviousScreen();
116             return true;
117         }
118 
119         return false;
120     }

122     /**
123      * This option is mainly for testing purposes. If the record store
124      * on the device or emulator gets into an unstable state, or has too 
125      * many images, you can reset it, which clears the record stores and
126      * re-creates them with the default images bundled with the application 
127      * @throws PersistenceMechanismException 
128      */
129     private void resetImageData() throws PersistenceMechanismException {
130         //begin - modified in MobileMedia-Cosmos-OO-v4
131         IManager manager = ComponentFactory.createInstance();
132         IFilesystem filesystem = (IFilesystem) manager.getRequiredInterface("IFilesystem");
133 
134         filesystem.resetImageData();
135 
136         //end - modified in MobileMedia-Cosmos-OO-v4
137 
138         //Clear the names from the album list
139         for (int i = 0; i < getAlbumListScreen().size(); i++) {
140             getAlbumListScreen().delete(i);
141         }
142 
143         //Get the default ones from the album
144         //begin - modified in MobileMedia-Cosmos-OO-v4
145         String[] albumNames = filesystem.getAlbumNames();
146         //end - modified in MobileMedia-Cosmos-OO-v4
147 
148         for (int i = 0; i < albumNames.length; i++) {
149             if (albumNames[i] != null) {
150                 //Add album name to menu list
151                 getAlbumListScreen().append(albumNames[i], null);
152             }
153         }
154         setCurrentScreen(getAlbumListScreen());
155     }

223     private void deletePhotoAlbum(){
224         System.out.println("Delete Photo Album here");
225         List down = (List) Display.getDisplay(midlet).getCurrent();
226         ScreenSingleton.getInstance().setCurrentScreenName(Constants.CONFIRMDELETEALBUM_SCREEN);
227         ScreenSingleton.getInstance().setCurrentStoreName(down.getString(down.getSelectedIndex()));
228         String message = "Would you like to remove the album "+ScreenSingleton.getInstance().getCurrentStoreName();
229         Alert deleteConfAlert = new Alert("Delete Photo Album", message,null,AlertType.CONFIRMATION);
230         deleteConfAlert.setTimeout(Alert.FOREVER);
231         deleteConfAlert.addCommand(new Command("Yes - Delete", Command.OK, 2));
232         deleteConfAlert.addCommand(new Command("No - Delete", Command.CANCEL, 2));
233         setAlbumListAsCurrentScreen(deleteConfAlert);
234         deleteConfAlert.setCommandListener(this);
235     }
