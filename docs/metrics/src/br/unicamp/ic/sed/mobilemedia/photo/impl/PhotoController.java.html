98      private boolean goToPreviousScreen() throws UnavailablePhotoAlbumException {
99          System.out.println("<* PhotoController.goToPreviousScreen() *>");
100         String currentScreenName = ScreenSingleton.getInstance().getCurrentScreenName();
101 
102         if (currentScreenName.equals(Constants.ALBUMLIST_SCREEN)) {
103             System.out.println("Can't go back here...Should never reach this spot");
104         } else if (currentScreenName.equals(Constants.IMAGE_SCREEN)) {          
105             //Go to the image list here, not the main screen...
106 
107             this.showImageList(this.getCurrentStoreName(), false, false);
108             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
109             return true;
110         }
111         else if (currentScreenName.equals(Constants.ADDPHOTOTOALBUM_SCREEN)) {
112 
113             this.showImageList(this.getCurrentStoreName(), false, false);
114             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
115             return true;
116         }else if(currentScreenName.equals(Constants.IMAGELIST_SCREEN)){
117             IManager manager = ComponentFactory.createInstance();
118             IMobilePhone phone = (IMobilePhone) manager.getRequiredInterface("IMobilePhone");
119             phone.goToPreviousScreen();
120             return true;
121         }else if(currentScreenName.equals(Constants.NEWLABEL_SCREEN )){
122             this.showImageList(this.getCurrentStoreName(), false, false);
123             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
124             return true;
125         }
126         System.out.println("[PhotoController could not handle either BACK or CANCEL: goToPreviousScreen() returns false]");
127         return false;
128     }

140     public boolean handleCommand(Command command) throws ImageNotFoundException, NullAlbumDataReference, PersistenceMechanismException, InvalidImageDataException, UnavailablePhotoAlbumException  {
141         String label = command.getLabel();
142         System.out.println( "<*"+this.getClass().getName()+".handleCommand() *> " + label);
143 
144         IManager manager = (IManager) ComponentFactory.createInstance();
145         IFilesystem filesystem = (IFilesystem) manager.getRequiredInterface("IFilesystem");
146         //IPhoto photo = (IPhoto) manager.getRequiredInterface("IPhoto");
147 
148         if (label.equals("View")) {
149             String selectedImageName = getSelectedImageName();
150             showImage(selectedImageName);
151 
152             // #ifdef includeSorting
153             // [EF] Added in the scenario 02
154 
155             IImageData image;
156 
157             
158             image = filesystem.getImageInfo(selectedImageName);
159             
160             image.increaseNumberOfViews();
161             this.updateImage(image);
162 
163             // #endif
164             //System.out.println("<* BaseController.handleCommand() *> Image = " + selectedImageName + "; # views = " + image.getNumberOfViews());
165 
166 
167             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGE_SCREEN);
168 
169             return true;
170 
171             /** Case: Add photo * */
172         } else if (label.equals("Add")) {
173             ScreenSingleton.getInstance().setCurrentScreenName(Constants.ADDPHOTOTOALBUM_SCREEN);
174             String storeName = this.getCurrentStoreName();
175             this.initAddPhotoToAlbum( storeName );
176             return true;
177 
178             /** Case: Save Add photo * */
179         } else if (label.equals("Save Photo")) {
180             //TODO print message error on the screen
181             
182                 AddPhotoToAlbum addPhotoToAlbum = (AddPhotoToAlbum) this.getCurrentScreen();
183                 String addedPhotoName = addPhotoToAlbum.getPhotoName();
184                 String addedPhotoPath = addPhotoToAlbum.getPath();
185                 filesystem.addNewPhotoToAlbum( addedPhotoName , addedPhotoPath , getCurrentStoreName() );
186             
187             /*getAlbumData().addNewPhotoToAlbum(((AddPhotoToAlbum) getCurrentScreen()).getPhotoName(), 
188                     ((AddPhotoToAlbum) getCurrentScreen()).getPath(), getCurrentStoreName());*/
189             return goToPreviousScreen();
190             /** Case: Delete selected Photo from recordstore * */
191         } else if (label.equals("Delete")) {
192             String selectedImageName = getSelectedImageName();
193             
194             filesystem.deleteImage(selectedImageName, getCurrentStoreName());
195             
196             showImageList(getCurrentStoreName(), false, false);
197             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
198             return true;
199 
200             /** Case: Edit photo label
201              *  [EF] Added in the scenario 02 */
202         } else if (label.equals("Edit Label")) {
203             this.editLabel();
204             return true;
205 
206             // #ifdef includeSorting
207             /**
208              * Case: Sort photos by number of views 
209              * [EF] Added in the scenario 02 */
210         } else if (label.equals("Sort by Views")) {
211             showImageList(getCurrentStoreName(), true, false);
212             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
213 
214             return true;
215             // #endif
216 
217             // #ifdef includeFavourites
218             /**
219              * Case: Set photo as favorite
220              * [EF] Added in the scenario 03 */
221         } else if (label.equals("Set Favorite")) {
222             String selectedImageName = getSelectedImageName();
223             
224                 //ImageData image = getAlbumData().getImageInfo(selectedImageName);
225                 IImageData image = filesystem.getImageInfo(selectedImageName);
226                 image.toggleFavorite();
227                 updateImage(image);
228                 System.out.println("<* BaseController.handleCommand() *> Image = "+ selectedImageName + "; Favorite = " + image.isFavorite());
229             
230             return true;
231 
232             /**
233              * Case: View favorite photos 
234              * [EF] Added in the scenario 03 */
235         } else if (label.equals("View Favorites")) {
236             showImageList(getCurrentStoreName(), false, true);
237             ScreenSingleton.getInstance().setCurrentScreenName(Constants.IMAGELIST_SCREEN);
238 
239             return true;
240             // #endif
241 
242             /** Case: Save new Photo Label */
243         } else if (label.equals("Save new label")) {
244             NewLabelScreen newLabelScreen = this.getScreen();
245             String labelName = newLabelScreen.getLabelName();
246             System.out.println("<* PhotoController.handleCommand() *> Save Photo Label = "+ labelName);
247             this.getImage().setImageLabel( labelName );
248             this.updateImage(image);
249             return this.goToPreviousScreen();
250 
251             /** Case: Go to the Previous or Fallback screen * */
252         } else if (label.equals("Back")) {
253             System.out.println("[PhotoController.handleCommand()] Back");
254             return this.goToPreviousScreen();
255 
256             /** Case: Cancel the current screen and go back one* */
257         } else if (label.equals("Cancel")) {
258             return this.goToPreviousScreen();
259 
260         }
261         
262         return false;
263     }

314     private void showImage(String name) throws ImageNotFoundException, PersistenceMechanismException {
315         // [EF] Instead of replicating this code, I change to use the method "getSelectedImageName()".      
316         Image storedImage = null;
317         IManager manager = ComponentFactory.createInstance();
318         IFilesystem filesystem = (IFilesystem) manager.getRequiredInterface("IFilesystem");
319 
320         storedImage = filesystem.getImageFromRecordStore(getCurrentStoreName(), name);
321 
322 
323         //We can pass in the image directly here, or just the name/model pair and have it loaded
324         PhotoViewScreen canv = new PhotoViewScreen(storedImage);
325         canv.setCommandListener( this );
326         AbstractController nextcontroller = this; 
327         // #if includeCopyPhoto || includeSmsFeature
328         PhotoViewController controller = new PhotoViewController(midlet, name);
329         controller.setNextController(nextcontroller);
330         canv.setCommandListener(controller);
331         nextcontroller = controller;
332         // #endif
333         
334         //#ifdef includeSmsFeature
335         /* [NC] Added in scenario 06 */
336         ISms sms = (ISms) manager.getRequiredInterface("ISms");
337         this.setNextController( sms );
338         //#endif
339         
340         setCurrentScreen(canv);
341     }
