9   package br.unicamp.ic.sed.mobilemedia.album.impl;
10  
11  import javax.microedition.lcdui.Command;
12  import javax.microedition.midlet.MIDlet;
13  
14  import br.unicamp.ic.sed.mobilemedia.album.spec.prov.IAlbum;
15  import br.unicamp.ic.sed.mobilemedia.album.spec.prov.IManager;
16  import br.unicamp.ic.sed.mobilemedia.album.spec.req.IMobileResources;
17  
18  
19  class IAlbumFacade implements IAlbum{
20  
21      private AlbumListScreen albumListScreen;
22      private AlbumController albumController;
23      
24      private AlbumListScreen getAlbumListScreen() {
25          this.albumListScreen = new AlbumListScreen();
26          return albumListScreen;
27      }
28  
29      private AlbumController getAlbumController() {
30          if( this.albumController == null){
31              IManager manager = ComponentFactory.createInstance();
32              IMobileResources mobileResources = (IMobileResources) manager.getRequiredInterface("IMobileResources");
33              MIDlet midlet = mobileResources.getMainMIDlet();
34              AlbumListScreen albumListScreen = this.getAlbumListScreen();
35              this.albumController = new AlbumController( midlet ,  albumListScreen );
36          }
37          return albumController;
38      }
39  
40  
41  
42      public void initAlbumListScreen ( String[] albumNames ){
43          //initialize all the screens
44          AlbumListScreen albumListScr = this.getAlbumListScreen();
45          for (int i = 0; i < albumNames.length; i++) {
46              if (albumNames[i] != null) {
47                  //Add album name to menu list
48                  albumListScr.append(albumNames[i], null);
49              }
50          }
51  
52          albumListScr.initMenu();
53  
54          //Set the current screen to the photo album listing
55          AlbumController albumCtr = this.getAlbumController();
56  
57          albumCtr.setCurrentScreen(albumListScr);
58  
59          albumListScr.setCommandListener( albumCtr );
60  
61      } 
62  
63      public boolean postCommand(Command command){
64          return false;
65      }
